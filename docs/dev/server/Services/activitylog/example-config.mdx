import yaml from 'js-yaml';
import configYamlRaw from '@site/static/env-vars/activitylog-config-example.yaml?raw';
import { useMemo } from 'react';
import CodeBlock from '@theme/CodeBlock';

export default function ConfigYamlPage() {
  const parsed = useMemo(() => yaml.load(configYamlRaw), []);

  return (
    <>
      <h1>Example Configuration </h1>

      <h2>Rohdaten</h2>
      <CodeBlock language="yaml">{configYamlRaw}</CodeBlock>

      <h2>Geparst</h2>
      <CodeBlock language="json">{JSON.stringify(parsed, null, 2)}</CodeBlock>
    </>
  );
}
